252 inline void System::computeForceDrive()
253 {
254  // compute the average displacement per layer
255  // (skip all layers that are not driven, the average displacement will never be used)
256 
257  m_layer_ubar_value.fill(0.0);
258  size_t nip = m_quad.nip();
259 
260  m_vector.asElement(m_u, m_ue);
261  m_quad.interpQuad_vector(m_ue, m_uq);
262 
263  for (size_t i = 0; i < m_n_layer; ++i) {
264  for (size_t d = 0; d < 2; ++d) {
265  if (m_layer_ubar_set(i, d)) {
266  for (auto& e : m_layer_elem[i]) {
267  for (size_t q = 0; q < nip; ++q) {
268  m_layer_ubar_value(i, d) += m_uq(e, q, d) * m_dV(e, q);
269  }
270  }
271  }
272  }
273  }
274 
275  m_layer_ubar_value /= m_layer_dV1;
276 
277  // compute the driving force
278 
279  m_uq.fill(0.0);
280 
281  for (size_t i = 0; i < m_n_layer; ++i) {
282  for (size_t d = 0; d < 2; ++d) {
283  if (m_layer_ubar_set(i, d)) {
284  double f = m_k_drive * (m_layer_ubar_value(i, d) - m_layer_ubar_target(i, d));
285  if (m_drive_spring_symmetric || f < 0) { // buckle under compression
286  for (auto& e : m_layer_elem[i]) {
287  for (size_t q = 0; q < nip; ++q) {
288  m_uq(e, q, d) = f;
289  }
290  }
291  }
292  }
293  }
294  }
295 
296  m_quad.int_N_vector_dV(m_uq, m_ue);
297  m_vector.assembleDofs(m_ue, m_ud);
298  m_vector.asNode(m_ud, m_fdrive);
299 }
