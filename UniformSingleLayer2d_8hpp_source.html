132 inline double System::addSimpleShearEventDriven(
133  double deps_kick, bool kick, double direction, bool dry_run)
134 {
135  FRICTIONQPOTFEM_ASSERT(this->isHomogeneousElastic());
136  FRICTIONQPOTFEM_REQUIRE(direction == +1.0 || direction == -1.0);
137 
138  auto idx = this->plastic_CurrentIndex();
139  auto eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
140  auto Epsd = GMatTensor::Cartesian2d::Deviatoric(this->plastic_Eps());
141  auto epsxx = xt::view(Epsd, xt::all(), xt::all(), 0, 0);
142  auto epsxy = xt::view(Epsd, xt::all(), xt::all(), 0, 1);
143 
144  // distance to yielding: "deps"
145  // (event a positive kick can lead to a decreasing equivalent strain)
146  xt::xtensor<double, 2> sign = this->plastic_signOfSimpleShearPerturbation(direction * deps_kick);
147  xt::xtensor<double, 2> epsy_l = xt::abs(this->plastic_CurrentYieldLeft());
148  xt::xtensor<double, 2> epsy_r = xt::abs(this->plastic_CurrentYieldRight());
149  xt::xtensor<double, 2> epsy = xt::where(sign > 0, epsy_r, epsy_l);
150  xt::xtensor<double, 2> deps = xt::abs(eps - epsy);
151 
152  // no kick & current strain sufficiently close the next yield strain: don't move
153  if (!kick && xt::amin(deps)() < deps_kick / 2.0) {
154  return 0.0;
155  }
156 
157  // set yield strain close to next yield strain
158  xt::xtensor<double, 2> eps_new = epsy + sign * (-deps_kick / 2.0);
159 
160  // or, apply a kick instead
161  if (kick) {
162  eps_new = eps + sign * deps_kick;
163  }
164 
165  // compute shear strain increments
166  // - two possible solutions
167  // (the factor two is needed as "dgamma" is the xy-component of the deformation gradient)
168  xt::xtensor<double, 2> dgamma =
169  2.0 * (-1.0 * direction * epsxy + xt::sqrt(xt::pow(eps_new, 2.0) - xt::pow(epsxx, 2.0)));
170  xt::xtensor<double, 2> dgamma_n =
171  2.0 * (-1.0 * direction * epsxy - xt::sqrt(xt::pow(eps_new, 2.0) - xt::pow(epsxx, 2.0)));
172  // - discard irrelevant solutions
173  dgamma_n = xt::where(dgamma_n <= 0.0, dgamma, dgamma_n);
174  // - select lowest
175  dgamma = xt::where(dgamma_n < dgamma, dgamma_n, dgamma);
176  // - select minimal
177  double dux = xt::amin(dgamma)();
178 
179  if (dry_run) {
180  return direction * dux;
181  }
182 
183  // add as affine deformation gradient to the system
184  for (size_t n = 0; n < m_nnode; ++n) {
185  m_u(n, 0) += direction * dux * (m_coor(n, 1) - m_coor(0, 1));
186  }
187  this->updated_u();
188 
189  // sanity check
190  // ------------
191 
192  auto index = xt::unravel_index(xt::argmin(dgamma)(), dgamma.shape());
193  size_t e = index[0];
194  size_t q = index[1];
195 
196  eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
197  auto idx_new = this->plastic_CurrentIndex();
198 
199  FRICTIONQPOTFEM_REQUIRE(std::abs(eps(e, q) - eps_new(e, q)) / eps_new(e, q) < 1e-4);
200  if (!kick) {
201  FRICTIONQPOTFEM_REQUIRE(xt::all(xt::equal(idx, idx_new)));
202  }
203 
204  return direction * dux;
205 }

262 inline double System::triggerElementWithLocalSimpleShear(
263  double deps_kick,
264  size_t plastic_element,
265  bool trigger_weakest,
266  double amplify)
267 {
268  FRICTIONQPOTFEM_ASSERT(plastic_element < m_nelem_plas);
269 
270  auto idx = this->plastic_CurrentIndex();
271  auto eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
272  auto up = m_vector.AsDofs_p(m_u);
273 
274  // distance to yielding on the positive side
275  auto epsy = this->plastic_CurrentYieldRight();
276  auto deps = epsy - eps;
277 
278  // find integration point closest to yielding
279  size_t e = m_elem_plas(plastic_element);
280  size_t q = xt::argmin(xt::view(deps, plastic_element, xt::all()))();
281  if (!trigger_weakest) {
282  q = xt::argmax(xt::view(deps, plastic_element, xt::all()))();
283  }
284 
285  // deviatoric strain at the selected quadrature-point
286  xt::xtensor<double, 2> Eps = xt::view(this->plastic_Eps(), plastic_element, q);
287  xt::xtensor<double, 2> Epsd = GMatTensor::Cartesian2d::Deviatoric(Eps);
288 
289  // new equivalent deviatoric strain: yield strain + small strain kick
290  double eps_new = epsy(plastic_element, q) + deps_kick / 2.0;
291 
292  // convert to increment in shear strain (N.B. "dgamma = 2 * Epsd(0,1)")
293  double dgamma = 2.0 * (-Epsd(0, 1) + std::sqrt(std::pow(eps_new, 2.0) - std::pow(Epsd(0, 0), 2.0)));
294 
295  // apply increment in shear strain as a perturbation to the selected element
296  // - nodes belonging to the current element, from connectivity
297  auto elem = xt::view(m_conn, e, xt::all());
298  // - displacement-DOFs
299  auto udofs = m_vector.AsDofs(m_u);
300  // - update displacement-DOFs for the element
301  for (size_t n = 0; n < m_nne; ++n) {
302  udofs(m_dofs(elem(n), 0)) += dgamma * amplify * (m_coor(elem(n), 1) - m_coor(elem(0), 1));
303  }
304  // - convert displacement-DOFs to (periodic) nodal displacement vector
305  // (N.B. storing to nodes directly does not ensure periodicity)
306  this->setU(m_vector.AsNode(udofs));
307 
308  eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
309  auto idx_new = this->plastic_CurrentIndex();
310  auto up_new = m_vector.AsDofs_p(m_u);
311 
312  FRICTIONQPOTFEM_REQUIRE(dgamma >= 0.0);
313  FRICTIONQPOTFEM_REQUIRE(amplify >= 0.0);
314  FRICTIONQPOTFEM_REQUIRE(std::abs(eps(plastic_element, q) - eps_new) / eps_new < 1e-4 || amplify != 1);
315  FRICTIONQPOTFEM_REQUIRE(xt::any(xt::not_equal(idx, idx_new)));
316  FRICTIONQPOTFEM_REQUIRE(idx(plastic_element, q) != idx_new(plastic_element, q));
317  FRICTIONQPOTFEM_REQUIRE(xt::allclose(up, up_new));
318 
319  return dgamma;
320 }

556 inline void LocalTriggerFineLayerFull::setState(
557  const xt::xtensor<double, 4>& Eps,
558  const xt::xtensor<double, 4>& Sig,
559  const xt::xtensor<double, 2>& epsy,
560  size_t N)
561 {
562  FRICTIONQPOTFEM_ASSERT(xt::has_shape(Eps, m_shape_T2));
563  FRICTIONQPOTFEM_ASSERT(xt::has_shape(Sig, m_shape_T2));
564  FRICTIONQPOTFEM_ASSERT(xt::has_shape(epsy, {m_nelem_plas, m_nip}));
565 
566  xt::xtensor<double, 2> S = xt::empty<double>({size_t(2), N});
567  xt::xtensor<double, 2> P = xt::empty<double>({size_t(2), N});
568  xt::xtensor<double, 2> W = xt::empty<double>({size_t(2), N});
569 
570  for (size_t e = 0; e < m_nelem_plas; ++e) {
571 
572  auto Eps_s = this->Eps_s(e);
573  auto Eps_p = this->Eps_p(e);
574  auto Sig_s = this->Sig_s(e);
575  auto Sig_p = this->Sig_p(e);
576  auto Sig_slice = this->slice(Sig, e);
577  auto dV_slice = this->slice(m_dV, e);
578 
579  for (size_t q = 0; q < m_nip; ++q) {
580 
581  double dgamma = m_dgamma(e, q);
582  double dE = m_dE(e, q);
583 
584  auto Epsd = GMatTensor::Cartesian2d::Deviatoric(xt::eval(xt::view(Eps, m_elem_plas(e), q)));
585  double gamma = Epsd(0, 1);
586  double E = Epsd(0, 0);
587  double y = epsy(e, q);
588  double a, b, c, D;
589 
590  // solve for "p = 0"
591  a = SQR(dgamma);
592  b = 2.0 * gamma * dgamma;
593  c = SQR(gamma) + SQR(E) - SQR(y);
594  D = SQR(b) - 4.0 * a * c;
595  double smax = (- b + std::sqrt(D)) / (2.0 * a);
596  double smin = (- b - std::sqrt(D)) / (2.0 * a);
597 
598  // solve for "s = 0"
599  a = SQR(dE);
600  b = 2.0 * E * dE;
601  c = SQR(E) + SQR(gamma) - SQR(y);
602  D = SQR(b) - 4.0 * a * c;
603  double pmax = (- b + std::sqrt(D)) / (2.0 * a);
604  double pmin = (- b - std::sqrt(D)) / (2.0 * a);
605 
606  size_t n = static_cast<size_t>(- smin / (smax - smin) * static_cast<double>(N));
607  size_t m = N - n;
608 
609  for (size_t i = 0; i < 2; ++i) {
610  xt::view(S, i, xt::range(0, n)) = xt::linspace<double>(smin, 0, n);
611  xt::view(S, i, xt::range(n, N)) = xt::linspace<double>(smax / double(m), smax, m);
612  P(i, 0) = 0.0;
613  P(i, N - 1) = 0.0;
614  }
615  P(0, n - 1) = pmax;
616  P(1, n - 1) = pmin;
617 
618  for (size_t j = 1; j < N - 1; ++j) {
619  if (j == n - 1) {
620  continue;
621  }
622  a = SQR(dE);
623  b = 2.0 * E * dE;
624  c = SQR(E) + std::pow(gamma + S(0, j) * dgamma, 2.0) - SQR(y);
625  D = SQR(b) - 4.0 * a * c;
626  P(0, j) = (- b + std::sqrt(D)) / (2.0 * a);
627  P(1, j) = (- b - std::sqrt(D)) / (2.0 * a);
628  }
629 
630  for (size_t i = 0; i < P.shape(0); ++i) {
631  for (size_t j = 0; j < P.shape(1); ++j) {
632  xt::xtensor<double, 4> sig = P(i, j) * Sig_p + S(i, j) * Sig_s + Sig_slice;
633  xt::xtensor<double, 4> deps = P(i, j) * Eps_p + S(i, j) * Eps_s;
634  xt::xtensor<double, 2> w = GMatTensor::Cartesian2d::A2s_ddot_B2s(sig, deps);
635  w *= dV_slice;
636  W(i, j) = xt::sum(w)();
637  }
638  }
639 
640  auto idx = xt::argmin(W);
641  m_smin(e, q) = S[idx];
642  m_pmin(e, q) = P[idx];
643  m_Wmin(e, q) = W[idx];
644  }
645  }
