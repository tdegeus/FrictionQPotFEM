471 inline void System::timeStep()
472 {
473  FRICTIONQPOTFEM_ASSERT(m_allset);
474 
475  // history
476 
477  m_t += m_dt;
478  xt::noalias(m_v_n) = m_v;
479  xt::noalias(m_a_n) = m_a;
480 
481  // new displacement
482 
483  xt::noalias(m_u) = m_u + m_dt * m_v + 0.5 * std::pow(m_dt, 2.0) * m_a;
484  this->updated_u();
485 
486  // estimate new velocity, update corresponding force
487 
488  xt::noalias(m_v) = m_v_n + m_dt * m_a_n;
489  this->updated_v();
490 
491  // compute residual force & solve
492 
493  this->computeInternalExternalResidualForce();
494  m_M.solve(m_fres, m_a);
495 
496  // re-estimate new velocity, update corresponding force
497 
498  xt::noalias(m_v) = m_v_n + 0.5 * m_dt * (m_a_n + m_a);
499  this->updated_v();
500 
501  // compute residual force & solve
502 
503  this->computeInternalExternalResidualForce();
504  m_M.solve(m_fres, m_a);
505 
506  // new velocity, update corresponding force
507 
508  xt::noalias(m_v) = m_v_n + 0.5 * m_dt * (m_a_n + m_a);
509  this->updated_v();
510 
511  // compute residual force & solve
512 
513  this->computeInternalExternalResidualForce();
514  m_M.solve(m_fres, m_a);
515 }

713 inline void HybridSystem::computeForceMaterial()
714 {
715  FRICTIONQPOTFEM_ASSERT(m_allset);
716  m_eval_full = true;
717 
718  m_vector_plas.asElement(m_u, m_ue_plas);
719  m_quad_plas.symGradN_vector(m_ue_plas, m_Eps_plas);
720  m_material_plas.setStrain(m_Eps_plas);
721  m_material_plas.stress(m_Sig_plas);
722  m_quad_plas.int_gradN_dot_tensor2_dV(m_Sig_plas, m_fe_plas);
723  m_vector_plas.assembleNode(m_fe_plas, m_fplas);
724 
725  m_K_elas.dot(m_u, m_felas);
726 
727  xt::noalias(m_fmaterial) = m_felas + m_fplas;
728 }
